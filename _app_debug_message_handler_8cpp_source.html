93 void AppDebugMessageHandler::messageHandler(QtMsgType type, const QMessageLogContext & context, const QString & msg)
94 {
95  static const char symbols[] = { 'D', 'W', 'E', 'X', 'I' }; // correspond to QtMsgType enum
96  // normalize types, QtDebugMsg stays 0, QtInfoMsg becomes 1, rest are QtMsgType + 1
97  quint8 lvl = type;
98  if (type == QtInfoMsg)
99  lvl = 1;
100  else if (type > QtDebugMsg)
101  ++lvl;
102 
103  if (lvl < m_appDebugOutputLevel && type != QtFatalMsg)
104  return;
105 
106 #if defined(Q_OS_LINUX) && (QT_VERSION < QT_VERSION_CHECK(5, 3, 0))
107  // Filter out lots of QPainter warnings from undocked QDockWidgets... hackish but effective (only workaround found so far)
108  if (lvl == 2 && QString(context.function).contains("QPainter::"))
109  return;
110 #endif
111 
112  QString msgPattern = QString("[%1] ").arg(symbols[type]);
113 
114  if (m_showSourcePath) {
115  QString file = QString("%1::").arg(context.file);
116  file.replace(m_srcPathFilter, "\\1");
117  msgPattern.append(file);
118  }
119  if (m_showFunctionDeclarations)
120  msgPattern.append(context.function);
121  else
122  msgPattern.append("%{function}()");
123 
124  msgPattern.append(":%{line} -%{if-category} [%{category}]%{endif} %{message}");
125 
126 #ifndef Q_OS_WIN
127  if (type == QtFatalMsg)
128  msgPattern.append("\nBACKTRACE:\n%{backtrace depth=12 separator=\"\n\"}");
129 #endif
130 
131  qSetMessagePattern(msgPattern);
132 
133  if (!m_defaultHandler || m_outputDevices.size() || receivers(SIGNAL(messageOutput(quint8, const QString &)))) {
134 #if (QT_VERSION >= QT_VERSION_CHECK(5, 4, 0))
135  msgPattern = qFormatLogMessage(type, context, msg);
136 #else
137  msgPattern.replace("%{line}", QString::number(context.line));
138  msgPattern.replace("%{if-category} [%{category}]%{endif}", QString(context.category));
139  msgPattern.replace("%{message}", msg);
140  if (!m_showFunctionDeclarations) {
141  QString func = context.function;
142  msgPattern.replace("%{function}", func.replace(m_functionFilter, "\\1)"));
143  }
144 #endif
145 
146  emit messageOutput(lvl, msgPattern);
147 
148  foreach (QIODevice * d, m_outputDevices) {
149  if (d && d->isWritable() && (!d->property("level").isValid() || d->property("level").toInt() <= lvl)) {
150  d->write(qPrintable(msgPattern % "\n"));
151  if (QFileDevice * fd = qobject_cast<QFileDevice *>(d))
152  fd->flush();
153  }
154  }
155  }
156 
157  // if (QThread::currentThread() == qApp->thread()) // gui thread
158 
159  if (m_defaultHandler) {
160  m_defaultHandler(type, context, msg);
161  }
162  else {
163  fprintf(stderr, "%s", qPrintable(msgPattern));
164  if (type == QtFatalMsg)
165  abort();
166  }
167 }
