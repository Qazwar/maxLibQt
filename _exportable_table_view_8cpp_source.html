86 QString ExportableTableView::toPlainText(const QModelIndexList &indexList, const QString &delim) const
87 {
88  QString ret, header;
89  const QChar rowDelim = '\n';
90  bool firstRow = true;
91  for (int i = 0; i < indexList.count(); ++i) {
92  const QModelIndex & idx = indexList.at(i);
93 
94  if (firstRow)
95  header.append(model()->headerData(idx.column(), Qt::Horizontal).toString());
96 
97  ret.append(idx.data(Qt::DisplayRole).toString());
98 
99  if (i + 1 == indexList.count() || indexList.at(i+1).row() != idx.row()) {
100  ret.append(rowDelim);
101  if (firstRow && !header.isEmpty())
102  header.append(rowDelim);
103  firstRow = false;
104  }
105  else {
106  ret.append(delim);
107  if (firstRow && !header.isEmpty())
108  header.append(delim);
109  }
110  }
111  if (!header.isEmpty())
112  ret.prepend(header);
113 
114  return ret;
115 }

117 QString ExportableTableView::toHtml(const QModelIndexList &indexList) const
118 {
119  QString ret, header, row;
120  bool firstRow = true;
121 
122  for (int i = 0; i < indexList.count(); ++i) {
123  const QModelIndex & idx = indexList.at(i);
124  const int algn = (idx.data(Qt::TextAlignmentRole).isValid() ? idx.data(Qt::TextAlignmentRole).toInt() : (Qt::AlignLeft | Qt::AlignVCenter));
125  const QString fg = (idx.data(Qt::ForegroundRole).isValid() ? idx.data(Qt::ForegroundRole).value<QBrush>().color().name(QColor::HexRgb) : "initial");
126  const QString bg = (idx.data(Qt::BackgroundRole).isValid() ? idx.data(Qt::BackgroundRole).value<QBrush>().color().name(QColor::HexRgb) : "initial");
127  const QString ttl = (idx.data(Qt::ToolTipRole).isValid() ? idx.data(Qt::ToolTipRole).toString().replace("\"", "\"\"") : QString());
128 
129  QString fnt;
130  if (idx.data(Qt::FontRole).isValid()) {
131  const QFont font = idx.data(Qt::FontRole).value<QFont>();
132  fnt = "font: \"" % font.family() % "\" " % QString::number(font.pointSize()) % "pt;";
133  }
134 
135  if (firstRow)
136  header.append(QString("<th align='left'>%1</th>\n").arg(model()->headerData(idx.column(), Qt::Horizontal).toString()));
137 
138  row.append("<td style='color: %2; background-color: %3; %4' align='%5' valign='%6' %7>%1</td>\n"); // cell template
139  row = row.arg(idx.data(Qt::DisplayRole).toString()).arg(fg).arg(bg).arg(fnt);
140  row = row.arg((algn & Qt::AlignRight) ? "right" : (algn & Qt::AlignHCenter) ? "center" : "left");
141  row = row.arg((algn & Qt::AlignTop) ? "top" : (algn & Qt::AlignBottom) ? "bottom" : "middle");
142  row = row.arg(ttl.isEmpty() ? ttl : QString("title=\"%1\"").arg(ttl));
143 
144  if (i + 1 == indexList.count() || indexList.at(i+1).row() != idx.row()) {
145  ret.append(QString("<tr>\n%1</tr>\n").arg(row));
146  row.clear();
147  firstRow = false;
148  }
149  }
150 
151  ret = QString("<tbody>\n%1</tbody>\n").arg(ret);
152  if (!header.isEmpty())
153  ret.prepend(QString("<thead>\n<tr>\n%1</tr>\n</thead>\n").arg(header));
154  ret = m_htmlTemplate.arg(ret);
155 
156  return ret;
157 }

159 bool ExportableTableView::saveToFile(const QModelIndexList &indexList, const QString &fileName)
160 {
161  static QString lastDir = QStandardPaths::writableLocation(QStandardPaths::DocumentsLocation);
162 
163  if (indexList.isEmpty())
164  return false;
165 
166  QString fname = fileName;
167  if (fname.isEmpty()) {
168  QString types = tr("Tab-delimited text") % " (*.tab);;" % tr("Comma-delimited text") % " (*.csv);;" % tr("Pipe-delimited text") % " (*.txt);;" % tr("HTML") % " (*.html)";
169  fname = QFileDialog::getSaveFileName(this, tr("Save to file"), lastDir, types);
170  }
171 
172  if (fname.isEmpty())
173  return false;
174 
175  QFile file(fname);
176  QFileInfo fi(file);
177  lastDir = fi.absolutePath();
178 
179  if (!file.open(QIODevice::WriteOnly | QIODevice::Text))
180  return false;
181 
182  if (fi.suffix() == "html")
183  file.write(toHtml(indexList).toUtf8());
184  else
185  file.write(toPlainText(indexList, (fi.suffix() == "tab" ? "\t" : fi.suffix() == "csv" ? ", " : " | ")).toUtf8());
186  file.close();
187 
188  return true;
189 }
